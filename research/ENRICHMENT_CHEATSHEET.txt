🎯 COMMON USE CASES

# Enrich specific accelerator
python scripts/universal_enrichment.py \
    --csv data/companies.csv \
    --filter-source "Y Combinator"

# Enrich specific batch
python scripts/universal_enrichment.py \
    --csv data/companies.csv \
    --filter-batch "W24"

# Save to separate file (don't modify main DB)
python scripts/universal_enrichment.py \
    --csv data/companies.csv \
    --output data/results.csv \
    --no-db

# Run in background
nohup python scripts/universal_enrichment.py \
    --csv data/companies.csv \
    --no-db > log.txt 2>&1 &
